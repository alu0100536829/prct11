        raise unless b.is_a?(Numeric)

def * (other)

        b = @b * other.denom
        k = gcd(a,b)
        a /= k
        b /= k
        Fraccion.new(a,b)

        a2 = other.num * @b

def -@
        k = gcd(@a, @b)
a = @a / k
b = @b / k

                return Fraccion.new(a,-b)
        else

        end
